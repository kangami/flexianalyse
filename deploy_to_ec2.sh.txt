#!/bin/bash

#Variables
EC2_USER="ec2-user"
EC2_IP="18.223.125.154"
PEM_FILE="/c/Users/KARIM NGAMI TEUMI/Downloads/flexianalyse.pem"
EC2_PATH="/var/www/html"

npm run build

  echo "❌ Échec du transfert des fichiers."
  exit 1

EOF
